             LINEAR SEARCH
    
#include<stdio.h>
int main(){
int a[10],i,item,n;
printf("\n enter number of elements of an array");
scanf("%d",&n);
printf("\n enter elements \n");
for(i=0;i<n;i++)
scanf("%d",&a[i]);
printf("\n enter item to search");
scanf("%d",&item);
for(i=0;i<=9;i++);
if(item==a[i]){
printf("\n item found at location %d",i+1);
break;
}
if(i>9)
printf("\n item does not exist");
return 0;
}
